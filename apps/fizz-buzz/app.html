/*
* Description
* The getValues function is called directly from the page using a 'click' event listener on the submit button.
*/
function getValues()
{
    // clears alert
    getElementBy("id", "alert").classList.add("d-none", "invisible");
    // Resets table information
    getElementBy("id", "tableData").innerHTML = "";

    //Set fizz and buzz input values to variables and attempt to parse into integers
    let fizzValue = getElementBy("id", "fizzValue").value;
    fizzValue = parseInt(fizzValue);

    let buzzValue = getElementBy("id", "buzzValue").value;
    buzzValue = parseInt(buzzValue);

    //Sets name for Fizz and Buzz that can also be used as class names for CSS styling
    let fizzText = "Fizz";
    let buzzText = "Buzz";

    //Check whether parsed values are integers if not display alert to user
    if (Number.isInteger(fizzValue) && Number.isInteger(buzzValue))
    {
        //Create and populate an array with FizzBuzz values which can then be displayed in a adjustable table without having to recreate the array each time
        displayArray = createArray(fizzValue, buzzValue, fizzText, buzzText);

        //Populate table with array data and remove hidden tags from results table
        displayTable(fizzText, buzzText);

        //Remove hidden class from table and table info
        getElementBy("id", "tableData").classList.remove("invisible", "d-none");
        getElementBy("id", "tableInfo").classList.remove("invisible", "d-none");

        // jump to results
        window.location.href = "#Results";
    }
    else
    {
        getElementBy("id", "tableInfo").classList.add("invisible", "d-none");
        getElementBy("id", "alert").classList.remove("d-none", "invisible");
    }
}
                                
                            
                                
/*
* Description
* Create and returns an array of 100 numbers substituting values for Fizz or Buzz where appropriate.
* 1. The function takes in four parameters:
*   a. two integers for the "fizz" and "buzz" values
*   b. two strings for the text to display when those values are present.
* 2. An empty array is made to store the formatted numbers.
* 3. The array is populated by looping over 100 numbers and using a ternary operator is substituted where appropriate.
*   a. If it is divisible by the value of Fizz or Buzz, they are substituted by either Fizz, Buzz or FizzBuzz.
*   b. If neither is substituted the value of the array is used.
* 4. The array is returned into a variable to display using a table template generator function.
*/
function createArray(fizzValue, buzzValue, fizzText, buzzText)
{
    let array = [];

    // Loop through 100 numbers an push into array either number of FizzBuzz value
    for (let i = 1;i <= 100;i++) {

        // Add Fizz, Buzz, or Both if numbers are divisible
        // If neither Fizz or Buzz is added set value to number of index
        let value = ((i % fizzValue == 0 ? fizzText : "") + (i % buzzValue == 0 ? buzzText : "") || i);

        array.push(value);
    }

    return array;
}
                                
                            
                                
/*
* Description
* Displays FizzBuzz array in a table of numbers with special formatting.
* 1. Maps an array of numbers, a string for "fizz", and a string for "buzz".
* 2. Resets the HTML table to start with an empty string.
* 3. Sets up variables for the start and end of each row of the table
* 4. sets number of columns based on a slider element in the HTML.
* 5. Determines the appropriate table row and color class based on array values.
* 6. Appends the formatted number to the tableHTML string.
* 7. Finally, tableHTML is used to update the table element in the HTML document.
*/
function displayTable(fizzName, buzzName)
{
    //Reset Table
    let tableHtml = "";

    //Set table row tags
    let tableRowStart = "";
    let tableRowEnd = "";

    //used for table row numbers
    let columns = getElementBy("id", "tableSlider").value;
    let iCountNum = 1;    

    //Value that will be displayed
    let fizBuzzValue = ''

    //sets colour class for table entries
    let colorClass = "";

    
    displayArray.map((element, index) => {

        //sets the table row numbers
        if (columns == 1)
        {
            tableRowStart = "";
            tableRowEnd = "";
        }
        else if (iCountNum == 1)
        {
            tableRowStart = "";
            tableRowEnd = "";
            iCountNum++;
        }
        else if (iCountNum >= columns)
        {
            tableRowStart = "";
            tableRowEnd = "";
            iCountNum = 1;
        }
        else
        {
            tableRowStart = "";
            tableRowEnd = "";
            iCountNum++;
        }

        //Set CSS class
        colorClass = element;
        
        if (element == fizzName + buzzName) {
            fizBuzzValue = `${element}`;
            colorClass = "";
        } else if (element == fizzName){
            fizBuzzValue = element;
        } else if (element == buzzName){
            fizBuzzValue = element;
        } else {
            fizBuzzValue = element;
        }

        tableHtml += `${tableRowStart}${fizBuzzValue}${tableRowEnd}`;

    })
    
    getElementBy("id", "tableData").innerHTML = tableHtml;
}
Helper Functions
                                
    /*
    * Description
    * A helper function that I use to shorten targeting a DOM element
    * 1. Takes in the type and name of an element and returns it into a variable
    * eg: let fizzValue = getElementBy('id', "fizzValue").value;
    */
    function getElementBy(type, value)
    {
        if(type === "id" || type === "id")
            return document.getElementById(value);
        
        if(type === 'class' || type === 'Class' || type === 'ClassName' || type === 'className')
            return document.getElementsByClassName(value);
        
        if(type === 'tag' || type === 'Tag' || type === 'TagName' || type === 'tagName')
            return document.getElementsByTagName(value);
    }
